93 template <typename T>
94 struct FromCharsConverter<T, typename std::enable_if_t<std::is_arithmetic_v<T>, void>>
95 {
96 static ConvertResult<T> Convert(const DIRE_STRING_VIEW& pChars)
97 {
98 T value;
99 if constexpr (std::is_same_v<T, bool>)
100 {
101 // bool is an arithmetic type but from_chars doesn't support it so we need a special case to handle it
102 if (pChars != "true" && pChars != "false" && pChars != "1" && pChars != "0")
103 {
104 return ConvertResult<T>(pChars, std::errc::invalid_argument);
105 }
106
107 value = (pChars == "true" || pChars == "1");
108 }
109 else
110 {
111 auto [unusedPtr, error] { ::std::from_chars(pChars.data(), pChars.data() + pChars.size(), value) };
112 (void)unusedPtr;
113 if (error != std::errc())
114 {
115 return ConvertResult<T>(pChars, error);
116 }
117 }
118
119 return value;
120 }
